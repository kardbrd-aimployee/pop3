    ```rust
        None,
        AddToBuilding { building_handle: ObjectHandle },
        RemoveFromBuilding { building_handle: ObjectHandle },
        DepositWood { building_handle: ObjectHandle, amount: u16 },
        SpawnAtBuilding { building_handle: ObjectHandle },
    }
    ```

    Modify `tick_state` to return `(TickResult, DeferredAction)` instead of just `TickResult`. Existing callers updated to ignore the DeferredAction (or pattern match on `DeferredAction::None`).

    **New state handlers** (add to person_state.rs):

    ```rust
    fn enter_enter_building(unit: &mut Unit) {
        // building_handle should already be set by command that triggered this state
        unit.state_timer = 0;
    }

    fn tick_enter_building(unit: &mut Unit) -> (TickResult, DeferredAction) {
        // Walk toward building position (handled by movement system)
        // When arrived (state_timer incremented by movement):
        unit.state_timer += 1;
        if unit.state_timer >= 30 { // arrival placeholder
            let action = if let Some(bh) = unit.building_handle {
                DeferredAction::AddToBuilding { building_handle: bh }
            } else {
                DeferredAction::None
            };
            (TickResult::StateComplete, action) // coordinator transitions to InsideBuilding
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_housing(unit: &mut Unit) {
        // Person is now housed
        unit.state_timer = 0;
    }

    fn tick_housing(_unit: &mut Unit) -> (TickResult, DeferredAction) {
        // Stay indefinitely until ejected
        (TickResult::Continue, DeferredAction::None)
    }

    fn enter_training(unit: &mut Unit) {
        // Training countdown set by coordinator based on conversion type
        // Default: 256 ticks for warrior conversion
        if unit.state_timer == 0 {
            unit.state_timer = 256;
        }
    }

    fn tick_training(unit: &mut Unit) -> (TickResult, DeferredAction) {
        if unit.state_timer > 0 {
            unit.state_timer -= 1;
        }
        if unit.state_timer == 0 {
            (TickResult::StateComplete, DeferredAction::None)
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_wait_outside(unit: &mut Unit) {
        let action_handle = unit.building_handle;
        unit.building_handle = None;
        unit.state_timer = 0;
    }

    fn tick_wait_outside(unit: &mut Unit) -> (TickResult, DeferredAction) {
        unit.state_timer += 1;
        if unit.state_timer >= 20 {
            (TickResult::StateComplete, DeferredAction::None)
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_gathering(unit: &mut Unit) {
        unit.state_timer = 0;
        unit.wood_carried = 0;
    }

    fn tick_gathering(unit: &mut Unit) -> (TickResult, DeferredAction) {
        // Walk toward tree target (movement system)
        unit.state_timer += 1;
        if unit.state_timer >= 40 { // arrived at tree
            (TickResult::StateComplete, DeferredAction::None) // -> GatheringWood
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_gathering_wood(unit: &mut Unit) {
        unit.state_timer = 60; // chop time
    }

    fn tick_gathering_wood(unit: &mut Unit) -> (TickResult, DeferredAction) {
        if unit.state_timer > 0 { unit.state_timer -= 1; }
        if unit.state_timer == 0 {
            unit.wood_carried = 1; // picked up wood
            (TickResult::StateComplete, DeferredAction::None) // -> CarryingWood
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_carrying_wood(unit: &mut Unit) {
        unit.state_timer = 0;
    }

    fn tick_carrying_wood(unit: &mut Unit) -> (TickResult, DeferredAction) {
        unit.state_timer += 1;
        if unit.state_timer >= 40 { // arrived at building
            let amount = unit.wood_carried;
            unit.wood_carried = 0;
            let action = if let Some(bh) = unit.building_handle {
                DeferredAction::DepositWood { building_handle: bh, amount }
            } else {
                DeferredAction::None
            };
            (TickResult::StateComplete, action) // -> Gathering (loop)
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn tick_drowning_ext(unit: &mut Unit) -> (TickResult, DeferredAction) {
        // Rapid health drain
        let drain = 50u16;
        // Note: health is on the ObjectHeader, not Unit directly.
        // Use state_timer to track drowning progress for now.
        unit.state_timer += 1;
        if unit.state_timer >= 20 {
            (TickResult::Dead, DeferredAction::None)
        } else {
            (TickResult::Continue, DeferredAction::None)
        }
    }

    fn enter_guard(unit: &mut Unit) {
        unit.guard_position = Some(WorldCoord::new(
            unit.position.x, unit.position.z
        ));
    }

    fn tick_guard(unit: &mut Unit) -> (TickResult, DeferredAction) {
        // Hold position - if displaced, walk back
        (TickResult::Continue, DeferredAction::None)
    }

    fn enter_dead(unit: &mut Unit) {
        unit.alive = false;
    }

    fn tick_dead(_unit: &mut Unit) -> (TickResult, DeferredAction) {
        (TickResult::Dead, DeferredAction::None)
    }
    ```
